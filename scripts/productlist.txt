db.createCollection( "productlist",
 {
   validator:
 { $jsonSchema:
 {
      bsonType: "object",
      required: [ "barcode","name","lastPrice" ],
      properties:
 {
         barcode: {
            bsonType: "string",
            description: "must be a string and is required"
         },
         name: {
            bsonType: "string",
            description: "must be a string and is required"
         },
         categoryName: {
            bsonType: "string"
         },
         lastPrice:{
                     bsonType:'string',
                     description:'must be a string and is required'
                            },
         lastModifiedTime: {
            bsonType: "date"
         },
         prices:{
                    bsonType:'array',
                    description:'must be an array and is required',
                    items:{
                        bsonType:'object',
                        required:['price','productUnit','insertDatetime'],
                        properties:{
                           
                            promotionPrice:{
                                bsonType:'string',
                                description:'must be a string and is required'
                            },
                            productUnit:{
                                bsonType:'string',
                                description:'must be an objectId and is required'
                            },
                           insertDatetime: {
                                bsonType: "date"
                           }
                        }
                    }
                }
      }
   } }
} )

db.productlist.createIndex( { "barcode": 1 }, { unique: true } )
